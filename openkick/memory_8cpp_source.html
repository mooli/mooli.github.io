123 char *Heap::allocate(size_t size) {
124  // essentially, we walk the memory list until we find the first sufficiently-large chunk, then
125  // carve what we need out of it and return the address.
126 
127  // cheap checks: if we were asked for no bytes, or this chunk doesn't have sufficient free
128  // space, we immediately bail
129  if(!size || size > this->free) return nullptr;
130 
131  size = (size + 7) & ~7; // round up to the next-largest multiple of 8 bytes
132 
133  // The pointer to the location that contains the pointer to the next memchunk. Essentially the
134  // previous node in the singly-linked list. We need the address of the pointer and not the
135  // pointer itself because we sometimes need to update it.
136  Chunk **pchunk = &this->first;
137  Chunk **found = nullptr;
138  while(*pchunk) { // While the next pointer is not nullptr, i.e. there are still memchunks
139  if((*pchunk)->size >= size) {
140  found = pchunk;
141  break;
142  }
143  pchunk = &(*pchunk)->next;
144  }
145 
146  // search unsuccessful, so fail the allocation
147  if(!found)
148  return nullptr;
149 
150  Chunk *chunk = *found;
151  char *cstart = reinterpret_cast<char *>(chunk);
152  this->free -= size;
153 
154  if(chunk->size == size) {
155  // we've found a chunk that's the exact size we need. Result!
156  *found = chunk->next; // unlink Chunk
157  return cstart;
158  } else {
159  // we've found a chunk that's larger than what we need, so it needs splitting. We carve what
160  // we require off the bottom and create a new Chunk at the top, updating the pointer to it.
161  *pchunk = new (cstart+size) Chunk (chunk->next, chunk->size-size);
162  return cstart;
163  }
164 }

223 char *Heap::allocate_at(char *memory, size_t size) {
224  // This is broadly similar to allocate, except that we allocate at a specific address, or not at
225  // all.
226 
227  // cheap checks: if we were asked for no bytes, or this chunk doesn't have sufficient free
228  // space, or we were asked to allocate nullptr, we immediately bail.
229  if(!memory || !size || size > this->free) return nullptr;
230 
231  size = (size + 7) & ~7; // round up to the next-largest multiple of 8 bytes
232 
233  Chunk **pchunk = &this->first;
234  while(*pchunk) { // While the next pointer is not nullptr, i.e. there are still memchunks
235  Chunk *chunk = *pchunk;
236  char *cstart = reinterpret_cast<char *>(chunk);
237  if(cstart <= memory && memory + size <= cstart + chunk->size) {
238  // We've found a chunk that completely contains the memory region we were looking to
239  // allocate. So we're in luck.
240  if(memory + size < cstart + chunk->size) {
241  // need to create new chunk between end of allocated block and end of the chunk, and
242  // link it in. This temporarily creates an overlap with "chunk" but that's OK as
243  // we're about to truncate or unlink that Chunk.
244  chunk->next = new (memory + size) Chunk(chunk->next, chunk->size - size);
245  }
246  if(cstart < memory) {
247  // need to truncate chunk
248  chunk->size -= size;
249  } else {
250  // need to unlink chunk
251  *pchunk = chunk->next;
252  }
253  this->free -= size;
254  return memory;
255  }
256  pchunk = &chunk->next;
257  };
258 
259  // we got to the end without finding the chunk
260  return nullptr;
261 }

271 void Heap::deallocate(char *memory, size_t size) {
272  // cheap checks: if we were asked to free nullptr, or no bytes, we immediately bail
273  if(!memory || !size) return;
274 
276  size = (size + 7) & ~7; // round up to the next-largest multiple of 8 bytes
277 
278  // update free space
279  this->free += size;
280 
281  // Deallocation isn't quite the reverse of allocation. In theory we can just insert a Chunk at
282  // \c memory and return, but that will leave the memory horribly fragmented. So we actually need
283  // to merge this proposed Chunk with the previous or subsequent Chunk that it touches.
284 
285  Chunk **pchunk = &this->first; // pointer to pointer to tested Chunk
286  Chunk *previous = nullptr; // previous Chunk, if there is any
287  // We want to find the Chunk just ahead of us.
288  while(*pchunk) { // While the next pointer is not nullptr, i.e. there are still memchunks
289  Chunk *chunk = *pchunk;
290  char *cstart = reinterpret_cast<char *>(chunk);
291  // right, we have found the Chunk after us, so we're done.
292  if(cstart > memory) break;
293  // Update pointers and go for another trip round
294  pchunk = &chunk->next;
295  previous = chunk;
296  }
297  // We now have \c previous pointing to the Chunk that needs to be immediately before us, or
298  // nullptr if it's the start of the list. \c *pchunk points to the Chunk that needs to be
299  // immediately after us, or nullptr if it's the end of the list.
300  Chunk *mc_next = *pchunk;
301 
303 
304  // If there's a node after us, we check if it's right after us, and merge with it if so.
305  if(mc_next) {
306  char *cnext = reinterpret_cast<char *>(mc_next);
307  if(cnext == memory + size) {
308  // blocks are right together, so merge.
309  size += mc_next->size;
310  mc_next = mc_next->next;
311  } else if(cnext < memory + size) {
313  return;
314  }
315  }
316 
317  // Similarly, if there's a node before us, we'll try and merge with that.
318  if(previous) {
319  // we might be able to merge with the previous chunk, so let's try that.
320  char *ptop = reinterpret_cast<char *>(previous) + previous->size;
321  if(ptop == memory) {
322  // blocks are right together, so merge.
323  previous->size += size;
324  previous->next = mc_next;
325  return;
326  } else if(ptop > memory) {
328  return;
329  }
330  }
331 
332  // Now create and link in a new free Chunk
333  *pchunk = new (memory) Chunk(mc_next, size);
334 }
