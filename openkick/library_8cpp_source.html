114  size_t count = 1 + (code & 0x0f);
115  if((code & 0xc0) == 0x40) { // repeat
116  switch(code & 0x30) {
117  case 0x00: { // repeat long
118  uint32_t v = *reinterpret_cast<const uint32_t *>(input);
119  input += sizeof(uint32_t);

453 void Library::sum_library(void) {
454  // don't bother summing a library if it doesn't want it
455  if(!flags & LIBF_SUMUSED)
456  return;
457  // now calculate the checksum
458  const size_t count = neg_size / sizeof(uint16_t);
459  const uint16_t *top = reinterpret_cast<uint16_t *>(this),
460  *p = top - count;
461  uint16_t newsum = 0;
462  while(p < top) newsum += *p++;
463  // is the library marked as changed? If so, we just update the checksum
464  if(flags & LIBF_CHANGED) {
465  flags = Flags(flags & ~LIBF_CHANGED);
466  } else if(sum && sum != newsum) {
468  execbase->Alert(0x81000003);
469  }
470  sum = newsum;
